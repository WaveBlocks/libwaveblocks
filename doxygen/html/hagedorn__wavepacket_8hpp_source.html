156  template<int N>
157  Eigen::Array<complex_t,1,N> evaluate_and_reduce_(const Eigen::Matrix<complex_t,D,N> &x) const
158  {
159  // determine number of quadrature points since template parameter N could be Eigen::Dynamic
160  int npts = x.cols();
161 
162  // use Kahan's algorithm to accumulate bases with O(1) numerical error instead of O(Sqrt(N))
163  KahanSum< Eigen::Array<complex_t,1,N> > psi( Eigen::Matrix<complex_t,1,N>::Zero(1,npts) );
164 
165  Evaluator<D,N> evaluator(eps_, parameters_, enumeration_, x);
166 
167  Eigen::Array<complex_t,Eigen::Dynamic,N> prev_basis(0,npts);
168  Eigen::Array<complex_t,Eigen::Dynamic,N> curr_basis(0,npts);
169  Eigen::Array<complex_t,Eigen::Dynamic,N> next_basis(1,npts);
170 
171  next_basis = evaluator.seed();
172 
173  psi += coefficients_[0]*next_basis.row(0).matrix();
174 
175  for (std::size_t islice = 0; islice < enumeration_->n_slices(); islice++) {
176  prev_basis = std::move(curr_basis);
177  curr_basis = std::move(next_basis);
178 
179  next_basis = evaluator.step(islice, prev_basis, curr_basis);
180 
181  for (long j = 0; j < next_basis.rows(); j++) {
182  complex_t cj = coefficients_[enumeration_->slice(islice+1).offset() + j];
183 
184  //prints: multi-index -> basis -> coefficient
185  //std::cout << enumeration_->slice(islice+1)[j] << " -> " << next_basis.row(j).matrix() << " * " << cj << std::endl;
186 
187  psi += cj*next_basis.row(j).matrix();
188  }
189  }
190 
191  return psi();
192  }

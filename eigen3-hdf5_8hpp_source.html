131 public:

134  {

136  assert(datatype->getSize() == sizeof(T));

278  template <typename Derived, typename DataSet>
279  void _load (const DataSet &dataset, const Eigen::DenseBase<Derived> &mat)
280  {
281  typedef typename Derived::Scalar Scalar;
282  const H5::DataSpace dataspace = dataset.getSpace();
283  const std::size_t ndims = dataspace.getSimpleExtentNdims();
284  assert(ndims > 0);
285  const std::size_t dimensions_size = 2;
286  hsize_t dimensions[dimensions_size];
287  dimensions[1] = 1; // in case it's 1D
288  if (ndims > dimensions_size) {
289  throw std::runtime_error("HDF5 array has too many dimensions.");
290  }
291  dataspace.getSimpleExtentDims(dimensions);
292  const hsize_t rows = dimensions[0], cols = dimensions[1];
293  std::vector<Scalar> data(rows * cols);
294  const H5::DataType * const datatype = DatatypeSpecialization<Scalar>::get();
295  internal::read_data(dataset, data.data(), *datatype);
296  // see http://eigen.tuxfamily.org/dox/TopicFunctionTakingEigenTypes.html
297  Eigen::DenseBase<Derived> &mat_ = const_cast<Eigen::DenseBase<Derived> &>(mat);
298  mat_.derived().resize(rows, cols);
299  mat_ = Eigen::Map<Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> >(data.data(), rows, cols);
300  }
